\newcommand{\addprereq}[1]{%
  \ifx\prereqlist\empty
    \def\prereqlist{#1}%
  \else
    \edef\prereqlist{\prereqlist, #1}%
  \fi
}

\NewDocumentCommand{\variosItems}{O{Ninguno}}{%
  \ifx\relax#1\relax
    Ninguno%
  \else
    \def\prereqlist{}%
    \forcsvlist{\addprereq}{#1}%
    \prereqlist
  \fi
}
