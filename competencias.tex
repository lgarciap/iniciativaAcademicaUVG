\subsection*{2.1 Genéricas}
% Seleccione máximo 3 de las 12 competencias genéricas que se listan a continuación y borre las demás.
\begin{multicols}{2}
    \begin{enumerate}[label={}]
        \item[1.] Piensa de manera crítica y analítica.
        \item[2.] Se comunica con efectividad.
        \item[3.] Trabaja en equipo.
        \item[4.] Resuelve problemas de manera creativa y efectiva.
        \item[5.] Utiliza adecuadamente la tecnología.
        \item[6.] Ejerce liderazgo.
        \item[7.] Actúa éticamente.
        \item[8.] Investiga en diferentes áreas del conocimiento.
        \item[9.] Practica la adaptabilidad en entornos cambiantes.
        \item[10.] Aprende a aprender autónomamente.
        \item[11.] Actúa en forma emprendedora e innovadora.
        \item[12.] Asume una postura intercultural y global.
    \end{enumerate}
\end{multicols}
